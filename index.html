function hasLink(id) { return !!(DRIVE_LINKS[id] && DRIVE_LINKS[id].trim()); }

function hasAnyLink() { return Object.values(DRIVE_LINKS).some(v => v && v.trim()); }


function openDoc(id) {

  const url = DRIVE_LINKS[id];

  if (url && url.trim()) { window.open(url, '_blank'); }

}


function cardHTML(doc) {

  const link = hasLink(doc.id);

  const indStr = doc.ind.slice(0,4).map(i=>`指標${i}`).join('、') + (doc.ind.length>4?' …':'');

  const phaseStr = doc.phase.join(' · ');

  return `<div class="doc-card ${link ? 'has-link' : ''}" onclick="${link ? `openDoc('${doc.id}')` : ''}">

    ${link ? '<span class="link-icon">↗</span>' : ''}

    <div class="card-top">

      <span class="tier-pill tier-${doc.tier}">${TIER_LABELS[doc.tier]}</span>

      <span class="card-name">${doc.id} — ${doc.name}</span>

    </div>

    <div class="card-meta">

      <span class="meta-tag">📋 ${phaseStr}</span>

      <span class="meta-tag">📌 ${indStr}</span>

    </div>

  </div>`;

}


function groupRender(groups) {

  return groups.map(({label, color, docs}) => {

    if (!docs.length) return '';

    return `<div class="section-group">

      <div class="section-head">

        <div class="color-bar" style="background:${color}"></div>

        ${label} <span style="font-weight:400;margin-left:4px">(${docs.length} 份)</span>

      </div>

      <div class="cards-grid">${docs.map(cardHTML).join('')}</div>

    </div>`;

  }).join('');

}


function applySearch(list) {

  if (!searchQ) return list;

  const q = searchQ.toLowerCase();

  return list.filter(d =>

    d.name.toLowerCase().includes(q) ||

    d.id.toLowerCase().includes(q) ||

    d.ind.some(i => String(i) === q || ('指標'+i).includes(q)) ||

    d.phase.some(p => p.includes(q)) ||

    d.pddro.some(s => s.toLowerCase().includes(q)) ||

    (TIER_LABELS[d.tier]||'').includes(q)

  );

}


function render() {

  const {type, value} = currentView;

  let filtered = applySearch(DOCS);

  let title = '全部文件';

  let html = '';


  if (type === 'phase') {

      html = `<div class="cards-grid">${filtered.map(cardHTML).join('')}</div>`;

    } else {

      title = '全部文件';

      html = groupRender([

        {label:'📋 課前作業', color:SECTION_COLORS['課前'], docs: filtered.filter(d=>d.phase.includes('課前'))},

    title = labels[value] || value;

    html = `<div class="cards-grid">${filtered.map(cardHTML).join('')}</div>`;

  } else if (type === 'indicator') {

    const ranges = {P:[1,2,3,4,5,6], D:[7,8,9,10,11,12,13,14], R:[15,16], O:[17,18,19]};

    const groups = inds.map(i => ({

      label: `指標 ${String(i).padStart(2,'0')} — ${IND_NAMES[i]}`,

      docs: filtered.filter(d => d.ind.includes(i))

    }));

    html = groupRender(groups);

}


function setView(type, value) {

  currentView = {type, value};

  searchQ = '';

  document.getElementById('sideSearch').value = '';

  document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));

  const key = type === 'phase' ? `nav-phase-${value}` : type === 'pddro' ? `nav-pddro-${value}` : type === 'indicator' ? `nav-ind-${value}` : `nav-tier-${value}`;

  const btn = document.getElementById(key);

  if (btn) btn.classList.add('active');

  render();

}


function handleSearch(val) {

  searchQ = val.trim();

  currentView = {type:'phase', value:'all'};

  document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));

  document.getElementById('nav-phase-all').classList.add('active');

  render();

}
